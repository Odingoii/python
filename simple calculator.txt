def calculator(num1, num2, operator)
 if operator== '+':
    return num1+num2
if operator== '-':
    return num1-num2
if operator== '/':
    return num1/num2
elif operator =='*':
    return num1*num2
num1 = float(input("Enter first number: "))
num2 = float(input("Enter second number: "))
operator = input("Enter operator : ")

print(calculator(num1, num2, operator))
